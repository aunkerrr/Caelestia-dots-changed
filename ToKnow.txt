# ==============================================================================
# Full optimization guide for Thinkpad E16 Gen 2 (Intel) + Arch + Hyprland + Caelestia Dots
# (Power plan menu is not going to work, handled automatically and efficiently by TLP)
# ==============================================================================

# Turning off default power profile  
# (Disabling the weak default power daemon to prevent conflicts with our custom settings)
sudo systemctl disable --now power-profiles-daemon
sudo systemctl mask power-profiles-daemon

# Installing main optimization dependencies
# (Installing TLP for advanced battery management and thermald to protect the CPU from overheating)
sudo pacman -S tlp thermald 

# Adding to autostartup 
# (Enabling their automatic startup on system boot)
sudo systemctl enable --now tlp.service
sudo systemctl enable --now thermald.service

# Open config 
# (Opening the TLP configuration file)
sudo nano /etc/tlp.conf

# Setting up TLP
# (Copy, paste, or uncomment the following lines in the tlp.conf file)

# --- Charging limits (scheme: 45% plugged in / 55% on battery) ---
# (Extending battery lifespan: preventing it from charging above 85% when plugged in)
START_CHARGE_THRESH_BAT0=80
STOP_CHARGE_THRESH_BAT0=90

# --- Taming Intel 13th Gen on battery (working on E-cores) ---
# (Disabling Turbo Boost and forcing the CPU to use efficient E-cores when disconnected from power)
CPU_SCALING_GOVERNOR_ON_BAT=powersave
CPU_ENERGY_PERF_POLICY_ON_BAT=balance_power
CPU_BOOST_ON_BAT=0

# --- Sleep for audio and Wi-Fi ---
# (Putting audio and Wi-Fi modules to sleep when they are idle)
SOUND_POWER_SAVE_ON_BAT=1
SOUND_POWER_SAVE_CONTROLLER=Y
WIFI_PWR_ON_BAT=on

# Apply TLP settings
# (Applying the settings immediately without rebooting)
sudo tlp start
